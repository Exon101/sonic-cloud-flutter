cmake_minimum_required(VERSION 3.14)
project(sonic_cloud LANGUAGES CXX)

set(BINARY_NAME "sonic_cloud")
set(APPLICATION_ID "com.example.sonic_cloud")

cmake_policy(SET CMP0078 NEW)
cmake_policy(SET CMP0147 NEW)

set(FLUTTER_MANIFEST_DIR "${CMAKE_CURRENT_LIST_DIR}/..")

add_compile_definitions(FLUTTER_VERSION="3.1.0+4")
add_compile_definitions(FLUTTER_VERSION_MAJOR=3)
add_compile_definitions(FLUTTER_VERSION_MINOR=1)
add_compile_definitions(FLUTTER_VERSION_PATCH=0)
add_compile_definitions(FLUTTER_VERSION_BUILD=4)

# Define apply_standard_settings as a macro — this is normally generated
# by `flutter create` but we define it here since our CMakeLists.txt is
# hand-written.
macro(apply_standard_settings target_name)
  set_target_properties(${target_name} PROPERTIES
    CXX_STANDARD 17
    CXX_STANDARD_REQUIRED ON
    CXX_VISIBILITY_PRESET hidden
    VISIBILITY_INLINES_HIDDEN YES
  )
  if(CMAKE_CXX_COMPILER_ID MATCHES "MSVC")
    target_compile_options(${target_name} PRIVATE /W4 /wd4100)
  elseif(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    target_compile_options(${target_name} PRIVATE -Wall -Wno-unused-parameter)
  endif()
endmacro()

# The Runner.rc.in file is normally generated by `flutter create`.
# Since we don't have it, skip the .rc resource file — the executable will
# still build, just without a Windows resource icon embedded.
if(EXISTS "${FLUTTER_MANIFEST_DIR}/windows/runner/Runner.rc.in")
  configure_file(
    "${FLUTTER_MANIFEST_DIR}/windows/runner/Runner.rc.in"
    "${CMAKE_CURRENT_BINARY_DIR}/Runner.rc"
  )
  set(RC_FILE "${CMAKE_CURRENT_BINARY_DIR}/Runner.rc")
else()
  set(RC_FILE "")
endif()

add_executable(${BINARY_NAME} WIN32
  "runner/main.cpp"
  "runner/flutter_window.cpp"
  "runner/utils.cpp"
  "runner/win32_window.cpp"
  ${RC_FILE}
  "runner/Runner.exe.manifest"
)

apply_standard_settings(${BINARY_NAME})
target_link_libraries(${BINARY_NAME} PRIVATE flutter flutter_wrapper_app)
target_include_directories(${BINARY_NAME} PRIVATE "${CMAKE_SOURCE_DIR}")

add_dependencies(${BINARY_NAME} flutter_assemble)
